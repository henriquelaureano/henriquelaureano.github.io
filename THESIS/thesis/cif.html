cif1 <- function(mrow) {
    t <- mrow[ , 1] ; eta1 <- mrow[ , 2] ; u1 <- mrow[ , 3] ; u2 <- 0
    X <- 1
    beta1 <- -2 ; beta2 <- -1
    risklevel1 <- exp(X * beta1 + u1)
    risklevel <- risklevel1/(1 + risklevel1 + exp(X * beta2 + u2))
    delta <- 80
    gamma1 <- 1
    w1 <- 3
    trajectory <- pnorm(w1 * atanh(2 * t/delta - 1) - X * gamma1 - eta1)
    out <- risklevel * trajectory
    return(out)
}
