data.inla <- list(y=y, x=x)

formula.rw1 <-
    y ~ -1 + f(x,
               model='rw1',
               scale.model=TRUE,
               hyper=list(prec=list(prior='pc.prec', param=c(1, 0.01))),
               constr=FALSE)
formula.ar1 <-
    y ~ -1 + f(x,
               model='ar1',
               hyper=list(theta1=list(prior='pc.prec', param=c(1, 0.01)),
                          theta2=list(prior='pc.cor1',
                                      param=c(0.1, 0.9))),
               constr=FALSE)
formula.spde <-
    y ~ -1 + f(x,
               model=inla.spde2.pcmatern(mesh=inla.mesh.1d(x), alpha=2,
                                         prior.range=c(1, 0.05),
                                         prior.sigma=c(1, 0.01)),
               constr=FALSE)

model.rw1  <- INLA::inla(formula.rw1, family='poisson', data=data.inla,
                         control.predictor=list(link=1))
model.ar1  <- INLA::inla(formula.ar1, family='poisson', data=data.inla,
                         control.predictor=list(link=1))
model.spde <- INLA::inla(formula.spde, family='poisson', data=data.inla,

data.plots <- function(model)
{
    f.hat <- model$summary.fitted.values$mean
    f.lb  <- model$summary.fitted.values$'0.025quant'
    f.ub  <- model$summary.fitted.values$'0.975quant'
    out   <- tibble::tibble(y=y, x=x, f.hat=f.hat, f.lb=f.lb, f.ub=f.ub)
    return(out)
}
