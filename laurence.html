data.plots <- function(model)
{
    f.hat <- model$summary.fitted.values$mean
    f.lb <- model$summary.fitted.values$'0.025quant'
    f.ub <- model$summary.fitted.values$'0.975quant'
    out <- tibble(
        y=y, x=x, f.hat=f.hat, f.lb=f.lb, f.ub=f.ub
    )
    return(out)
}
